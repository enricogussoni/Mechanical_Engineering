#include <stdio.h>

#define lunmax 10
#define max 10
#define min 0
int main()
{
int cont,n;
int numeri[lunmax],occorrenze[max-min];

                                                 /*Acquisizione numeri*/
for(n=0;n<lunmax;n++)
    {
	 printf("Inserisci il %d ° numero:\n",n+1);
	 scanf("%d",&numeri[n]);
	}

                                                    /*Conto occorrenze*/
for(cont=min;cont<=max;cont++)
    {
	 for(n=0;n<lunmax;n++)
	     {
		  if(numeri[n]==cont)
	         occorrenze[cont]=(occorrenze[cont]+1)
	     }
	}
	
return 0;
}
